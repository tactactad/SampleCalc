test('input after equal', function() {
    this.calc.inputValue(2);
    this.calc.inputOperator('add');
    this.calc.inputValue(3);

    this.calc.inputValue(4);
    deepEqual(this.calc.stack, '4');
    deepEqual(this.calc.operator, '');
